<script>
function calculateScore(){
  let total=0;
  for(let i=1;i<=16;i++){
    const radios=document.getElementsByName("q"+i);
    let val=null;
    for(const r of radios){if(r.checked) val=parseInt(r.value);}
    if(val===null){alert("رجاءً أجب على جميع الأسئلة");return;}
    total+=val;
  }
  let classification="";
  if(total<=32){classification="Low (استخدام طبيعي)"}
  else if(total<=55){classification="Moderate (علامات متوسطة)"}
  else{classification="High (Brain Rot شديد)"}

  const name=document.getElementById("username").value||"غير مذكور";
  const phone=document.getElementById("userphone").value||"غير مذكور";

  const res=document.getElementById("result");
  res.style.display="block";
  res.innerHTML=`<b>الاسم:</b> ${name}<br><b>الهاتف:</b> ${phone}<br><hr>
  <b>مجموع النقاط:</b> ${total}<br><b>التصنيف:</b> ${classification}`;
}
</script>
